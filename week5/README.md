    w0 = np.sum(prob[:t]) + 1e-8

    ‡∏Ñ‡∏≥‡∏ô‡∏ß‡∏ì œÉ¬≤_between(T) = w‚ÇÄ √ó w‚ÇÅ √ó (Œº‚ÇÄ - Œº‚ÇÅ)¬≤

def otsu(hist):
    prob = hist/np.sum(hist)
    coef_max, thr = -1, -1
    for t in range(1, 256):
        w0 = np.sum(prob[:t]); w1 = np.sum(prob[t:])
        u0 = np.sum(np.arange(t) * prob[:t]) / (w0 + 1e-8)
        u1 = np.sum(np.arange(t,256) * prob[t:]) / (w1 + 1e-8)
        coef = (w0*w1) * (u0 - u1)**2
        if coef > coef_max:
            coef_max, thr = coef, t
    return thr
